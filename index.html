EXP-5 Aim: To perform the system monitoring calculation -to calculate the frequency of pinging

clear clc;
wstn=24000;
data=128;
ip=0.05;
total_bits=((data*ip)+data)*8*wstn*2;
lan_rate=10^7;
capacity=0.3;
disp(total_bits);
time_taken=total_bits/lan_rate;
txtime=time_taken/capacity;
disp("txtime");
disp(txtime);
frequency=1/txtime;
disp("frequency");
disp(frequency);
i=1;
for capacity_b=0.05:0.01:0.5;
txtime_b(i)=time_taken/capacity_b;
frequency(i)=1/txtime_b(i);
disp(capacity_b);
j(i)=capacity_b;
i=i+1;
end
plot(j, frequency);
xlabel("efficiency");
ylabel("frequency");
